22 * @param value A pointer to the double value to animate.

50/**
51 * @brief Moves a value towards a target at a constant speed over a specified duration.
52 *
53 * This function is stateful. It tracks all ongoing linear animations and will
54 * correctly interpolate the value over the specified time 't'. If the target
55 * value changes for a given pointer, the animation will restart.
56 *
57 * @param value A pointer to the double value to animate.
58 * @param target The target value to animate towards.
59 * @param dt Delta time since the last frame.
60 * @param t The total time the animation should take in seconds.
61 */
62void gm_anim_linear(double *value, double target, const double dt,
63 const double t) {
64 if (value == NULL || t <= 0)
65 return;
66
67 // List of active linear animations.
68 static gmPtrList active_anims = NULL;
69
70 gmAnimLinearState *state = NULL;
71 int state_idx = -1;
72 size_t list_len = gm_ptr_list_length(active_anims);
73
74 // Find if an animation for this 'value' pointer already exists.
75 for (size_t i = 0; i < list_len; i++) {
76 gmAnimLinearState *s = (gmAnimLinearState *)active_anims[i];
77 if (s->value_ptr == value) {
78 state = s;
79 state_idx = i;
80 break;
81 }
82 }
83
84 // If the target has changed, we need to restart the animation.
85 if (state && state->target_value != target) {
86 // Free the old state and remove it from the list.
87 free(state);
88 active_anims = gm_ptr_list_pop_at(active_anims, state_idx);
89 state = NULL; // Force creation of a new state.
90 }
91
92 // If no state exists, create a new one.
93 if (state == NULL) {
94 // Don't start a new animation if it's already at the target.
95 if (*value == target)
96 return;
97
98 state = (gmAnimLinearState *)malloc(sizeof(gmAnimLinearState));
99 state->value_ptr = value;
100 state->start_value = *value;
101 state->target_value = target;
102 state->elapsed_time = 0.0;
103 state->total_time = t;
104 active_anims = gm_ptr_list_push(active_anims, state);
105 }
106
107 // Update the animation state.
108 state->elapsed_time += dt;
109
110 // Check for completion.
111 if (state->elapsed_time >= state->total_time) {
112 *value = state->target_value;
113 // Find the index again in case it was just added.
114 state_idx = gm_ptr_list_find(active_anims, state);
115 if (state_idx != -1) {
116 // Remove the state from the list and free memory.
117 active_anims = gm_ptr_list_pop_at(active_anims, state_idx);
118 }
119 free(state);
120 } else {
121 // Perform linear interpolation (lerp).
122 double fraction = state->elapsed_time / state->total_time;
123 *value = state->start_value +
124 (state->target_value - state->start_value) * fraction;
125 }
126}

171/**
172 * @brief Starts slow and accelerates quadratically towards the target.
173 * @param value A pointer to the double value to animate.
174 * @param target The target value to animate towards.
175 * @param dt Delta time since the last frame.
176 * @param t The animation's approximate duration.
177 */
178void gm_anim_ease_in_quad(double *value, double target, double dt, double t) {
179 if (value == NULL || t <= 0)
180 return;
181 double difference = target - *value;
182 if (fabs(difference) < 0.0001) {
183 *value = target;
184 return;
185 }
186 double speed_factor = 1.0 / (1.0 + sqrt(fabs(difference)));
187 double move = (dt * difference * speed_factor) / t;
188
189 // For ease-in, it's possible for the calculated move to be tiny, so we ensure
190 // minimum progress.
191 if (fabs(move) < 0.0001) {
192 move = 0.0001 * (difference > 0 ? 1 : -1);
193 }
194
195 if (fabs(move) >= fabs(difference)) {
196 *value = target;
197 } else {
198 *value += move;
199 }
200}
