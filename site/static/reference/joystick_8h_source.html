46/**
47 * @brief Creates and renders an animated joystick widget that can be manipulated with the mouse.
48 * @param x The x-coordinate of the joystick's center.
49 * @param y The y-coordinate of the joystick's center.
50 * @param radius The radius of the joystick base.
51 * @param pos Pointer to a gmPos structure to store the joystick's logical position (-1 to 1).
52 * @param vpos Pointer to a gmPos structure for animated visual position (can be NULL to use pos).
53 * @return 1 if the joystick is currently hovered, 0 otherwise.
54 */
55int gm_joystick_anim(double x, double y, double radius, gmPos *pos,
56 gmPos *vpos) {
57
58 if (!gmwJoystick.enabled)
59 return 0;
60 if (vpos == NULL)
61 vpos = pos;
62
63 // Mouse vector relative to center
64 double dx = gm_mouse.position.x - x;
65 double dy = gm_mouse.position.y - y;
66
67 double dist = sqrt(dx * dx + dy * dy);
68
69 int hovered = dist <= radius;
70
71 int active = hovered && gm_mouse.down;
72
73 if (!gm_mouse.down) {
74 gm_anim_spring(&pos->x, 0, gm_dt(), 0.1);
75 gm_anim_spring(&pos->y, 0, gm_dt(), 0.1);
76 }
77
78 // Logical joystick position (-1..1)
79 double nx = 0.0, ny = 0.0;
80
81 if (active) {
82 if (dist > radius) {
83 dx = dx / dist * radius;
84 dy = dy / dist * radius;
85 }
86 nx = dx / radius;
87 ny = dy / radius;
88
89 pos->x = nx;
90 pos->y = ny;
91 }
92
93 // Animate knob smoothly to target pos->x/pos->y
94 gm_anim_ease_out_quad(&vpos->x, pos->x, gm_dt(), 0.05);
95 gm_anim_ease_out_quad(&vpos->y, pos->y, gm_dt(), 0.05);
96
97 // Draw joystick circle (border + background)
98 double draw_radius = radius * gmwJoystick.scale;
99 gm_draw_circle(x, y, draw_radius + gmwJoystick.border_width,
100 gmwJoystick.border);
101 gm_draw_circle(x, y, draw_radius, gmwJoystick.background);
102
103 // Draw knob
104 double knob_radius = draw_radius * 0.3; // 30% of radius
105 gm_draw_circle(x + vpos->x * draw_radius, y + vpos->y * draw_radius,
106 knob_radius + gmwJoystick.border_width,
107 gmwJoystick.knob_border);
108 gm_draw_circle(
109 x + vpos->x * draw_radius, y + vpos->y * draw_radius, knob_radius,
110 nx == 0 && ny == 0 ? gmwJoystick.knob & 0xEEEEFF44 : gmwJoystick.knob);
111
112 return hovered;
113}
